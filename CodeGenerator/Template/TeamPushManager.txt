using System;
using System.Collections.Generic;
using System.Linq;
using System.Net.WebSockets;
using System.Threading.Tasks;
using GodModels;
using EClubServer.Services;

namespace GodNetwork
{
    public class TeamPushManager : PushManager
    {
        protected override void RegActionIds()
        {
#RegActionIdByType#
        }
        public override void AddPushAction(UserData userData, Action<GodPackage> callback)
        {
            TeamUserData tdata = userData as TeamUserData;
            if (!m_sockePusherById.ContainsKey(tdata.TeamId))
                m_sockePusherById[tdata.TeamId] = callback;
        }
        public override void RemovePushAction(UserData userData)
        {
            TeamUserData tdata = userData as TeamUserData;
            if (m_sockePusherById.ContainsKey(tdata.TeamId))
                m_sockePusherById.Remove(tdata.TeamId);
        }
    }
}
